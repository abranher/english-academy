model Platform {
  id          String   @id @default(uuid())
  name        String   // Nombre de la plataforma (ej: "EduHub", "LearnNow")
  description String?  @db.Text // Descripción general de la plataforma
  logoUrl     String?  // URL del logo de la plataforma
  primaryColor String? // Color primario de la plataforma (ej: para la interfaz de usuario)
  secondaryColor String? // Color secundario de la plataforma
  supportEmail String? // Correo electrónico de soporte
  supportPhone String? // Número de teléfono de soporte
  websiteUrl   String? // URL del sitio web principal
  termsOfServiceUrl String? // URL de los términos de servicio
  privacyPolicyUrl String? // URL de la política de privacidad
  aboutUsUrl String? // URL de la página "Acerca de nosotros"
  contactUsUrl String? // URL de la página "Contacto"
  facebookUrl String? // URL de la página de Facebook
  twitterUrl String? // URL de la página de Twitter
  instagramUrl String? // URL de la página de Instagram
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
}


model Platform {
  id                  String  @id @default(uuid())
  name                String  @unique
  description         String?
  websiteUrl          String?
  contactEmail        String? @unique
  socialMedia         Json?   // Ej: { twitter: "...", facebook: "...", instagram: "..." }
  commissionRate      Float?  // % que la plataforma retiene por transacciones
  maintenanceMode     Boolean @default(false)
  termsOfServiceUrl   String?
  privacyPolicyUrl    String?
  
  // Métricas automatizables (se actualizarían vía lógica de negocio)
  totalUsers          Int     @default(0)
  totalCourses        Int     @default(0)
  totalTransactions   Int     @default(0)

  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  @@unique([id]) // Garantiza que solo exista una instancia si se usa ID fijo
}
